#### A-1-2: C++ IOpHandler インターフェースを定義する

```cpp
// packages/native-engine/src/engine/IOpHandler.hpp

#pragma once
#include <string>
#include <vector>
#include <functional>

namespace kaedevn {

class IOpHandler {
public:
    virtual ~IOpHandler() = default;

    // --- テキスト ---
    virtual void textAppend(const std::string& who, const std::string& text) = 0;
    virtual void textNl() = 0;

    // --- 待機（ノンブロッキング: コールバックで完了通知） ---
    virtual void waitClick(std::function<void()> onComplete) = 0;
    virtual void page(std::function<void()> onComplete) = 0;
    virtual void waitMs(int ms, std::function<void()> onComplete) = 0;

    // --- 背景 ---
    virtual void bgSet(const std::string& id, int fadeMs, const std::string& effect,
                       float x, float y, float s) = 0;
    virtual void bgClear(int fadeMs) = 0;

    // --- キャラクター ---
    virtual void chSet(const std::string& name, const std::string& pose,
                       const std::string& pos, int fadeMs,
                       float x, float y, float s) = 0;
    virtual void chHide(const std::string& name, int fadeMs) = 0;
    virtual void chClear(int fadeMs) = 0;

    // --- 音声（即時、コールバック不要） ---
    virtual void bgmPlay(const std::string& id, float vol, int fadeMs) = 0;
    virtual void bgmStop(int fadeMs) = 0;
    virtual void sePlay(const std::string& id, float vol) = 0;
    virtual void voicePlay(const std::string& id) = 0;

    // --- 選択肢 ---
    virtual void showChoices(const std::vector<std::string>& labels,
                             std::function<void(int selected)> onComplete) = 0;

    // --- エフェクト（オプショナル — デフォルト実装は即時完了） ---
    virtual void shake(float intensity, int durationMs, std::function<void()> onComplete) {
        if (onComplete) onComplete();
    }
    virtual void flash(int durationMs, std::function<void()> onComplete) {
        if (onComplete) onComplete();
    }
    virtual void screenFilter(const std::string& filter, float intensity) {}
    virtual void screenFilterClear() {}

    // --- ViewState ---
    // セーブ/ロード用（後で拡張）
};

} // namespace kaedevn
```

**設計判断: コールバック方式**

TS の `Promise<void>` に相当するものとして、`std::function<void()> onComplete` コールバックを使う。

- C++20 コルーチンは Switch の開発環境で使えるか不明
- コールバックなら C++17 で確実に動く
- OpRunner 側で「onComplete が呼ばれるまで PC を進めない」制御をする

**完了条件:** ヘッダーが定義でき、SDL2Engine が（スタブでも）実装できること。

#### A-1-3: C++ OpRunner を実装する

```cpp
// packages/native-engine/src/engine/OpRunner.hpp

#pragma once
#include "Op.hpp"
#include "IOpHandler.hpp"

namespace kaedevn {

enum class RunnerState {
    Idle,
    Executing,
    WaitingCallback,  // Handler のコールバック待ち
    Finished
};

class OpRunner {
public:
    void start(const CompiledScenario& scenario, IOpHandler* handler);
    void resume(const CompiledScenario& scenario, IOpHandler* handler,
                int pc, const nlohmann::json& vars, const nlohmann::json& read);
    void stop();

    // メインループから毎フレーム呼ぶ
    // WaitingCallback でなければ次の Op を実行する
    void tick();

    RunnerState state() const { return state_; }
    int pc() const { return pc_; }
    const nlohmann::json& vars() const { return vars_; }
    const nlohmann::json& read() const { return read_; }

private:
    void executeNext();
    void onCommandComplete();  // コールバックから呼ばれる

    CompiledScenario scenario_;
    IOpHandler* handler_ = nullptr;
    RunnerState state_ = RunnerState::Idle;
    int pc_ = 0;
    nlohmann::json vars_;
    nlohmann::json read_;
};

} // namespace kaedevn
```

**tick() の実行フロー:**

```cpp
void OpRunner::tick() {
    // コールバック待ち or 終了なら何もしない
    if (state_ != RunnerState::Executing) return;

    // 同期 Op は連続実行（TEXT_APPEND, BGM_PLAY 等）
    // 非同期 Op（WAIT_CLICK, BG_SET 等）で止まる
    while (state_ == RunnerState::Executing && pc_ < scenario_.ops.size()) {
        executeNext();
    }

    if (pc_ >= scenario_.ops.size()) {
        state_ = RunnerState::Finished;
    }
}

void OpRunner::executeNext() {
    const auto& op = scenario_.ops[pc_];
    auto completeCb = [this]() { onCommandComplete(); };

    switch (op.type) {
        case OpType::TEXT_APPEND:
            handler_->textAppend(
                op.data.value("who", ""),
                op.data.value("text", "")
            );
            pc_++;
            break;  // 同期 → すぐ次へ

        case OpType::WAIT_CLICK:
            state_ = RunnerState::WaitingCallback;
            handler_->waitClick(completeCb);
            return;  // 非同期 → tick() を抜ける

        case OpType::BG_SET:
            handler_->bgSet(
                op.data.value("id", ""),
                op.data.value("fadeMs", 0),
                op.data.value("effect", "fade"),
                op.data.value("x", 0.0f),
                op.data.value("y", 0.0f),
                op.data.value("s", 1.0f)
            );
            pc_++;
            break;  // bgSet は状態を変えるだけ（アニメは update() で進行）

        case OpType::BGM_PLAY:
            handler_->bgmPlay(
                op.data.value("id", ""),
                op.data.value("vol", 1.0f),
                op.data.value("fadeMs", 0)
            );
            pc_++;
            break;  // 即時

        case OpType::CHOICE:
            state_ = RunnerState::WaitingCallback;
            {
                std::vector<std::string> labels;
                std::vector<int> jumps;
                for (const auto& opt : op.data["options"]) {
                    labels.push_back(opt.value("label", ""));
                    jumps.push_back(opt.value("jump", 0));
                }
                handler_->showChoices(labels, [this, jumps](int selected) {
                    if (selected >= 0 && selected < (int)jumps.size()) {
                        pc_ = jumps[selected];
                    }
                    onCommandComplete();
                });
            }
            return;

        case OpType::JUMP:
            pc_ = op.data.value("target", 0);
            break;

        case OpType::JUMP_IF: {
            const std::string varName = op.data.value("var", "");
            const auto& expected = op.data["value"];
            if (vars_.contains(varName) && vars_[varName] == expected) {
                pc_ = op.data.value("target", 0);
            } else {
                pc_++;
            }
            break;
        }

        case OpType::VAR_SET:
            vars_[op.data.value("var", "")] = op.data["value"];
            pc_++;
            break;

        default:
            // 未実装 Op はスキップ
            pc_++;
            break;
    }
}

void OpRunner::onCommandComplete() {
    read_[pc_] = true;
    pc_++;
    state_ = RunnerState::Executing;
}
```

**完了条件:**
- Op JSON を読んで TEXT_APPEND → WAIT_CLICK → BG_SET → CHOICE の基本フローが動く
- ユニットテスト: MockHandler で dispatch 順序を検証

// Before
void SDL2Engine::waitForClick() {
    if (skipping) return;
    bool clicked = false;
    while (!clicked) {
        while (SDL_PollEvent(&event)) { if (click) clicked = true; }
        render();
        SDL_Delay(16);
    }
}

// After
void SDL2Engine::waitClick(std::function<void()> onComplete) {
    if (skipping_) {
        onComplete();
        return;
    }
    waitingForClick_ = true;
    pendingCallback_ = onComplete;
    // ← ここで return。メインループに戻る
}

// main.cpp のイベント処理で:
void SDL2Engine::handleClick() {
    if (waitingForClick_ && pendingCallback_) {
        waitingForClick_ = false;
        auto cb = std::move(pendingCallback_);
        pendingCallback_ = nullptr;
        cb();  // OpRunner::onCommandComplete() が呼ばれる
    }
}

#### A-2-2: main.cpp をシングルループに書き換え

```cpp
// packages/native-engine/src/main.cpp（構造のみ）

int main(int argc, char* argv[]) {
    // --- 初期化 ---
    SDL_Init(...);
    auto window = SDL_CreateWindow("kaedevn", ...);
    auto renderer = createRenderer(window);  // GL or Soft
    auto audioManager = AudioManager();
    auto assetProvider = AssetProvider();
    auto inputManager = InputManager();

    // --- Op JSON 読み込み ---
    auto scenarioJson = loadJson(scriptPath);
    auto scenario = loadScenario(scenarioJson);

    // --- Handler + Runner 作成 ---
    auto handler = SDL2Handler(renderer, audioManager, assetProvider);
    auto runner = OpRunner();
    runner.start(scenario, &handler);

    // --- 入力バインド ---
    inputManager.on(Action::OK, [&]() {
        handler.handleClick();  // waitClick のコールバック発火
    });
    inputManager.on(Action::SkipToggle, [&]() {
        handler.toggleSkip();
    });
    inputManager.on(Action::Back, [&]() {
        handler.handleBack();
    });

    // --- メインループ ---
    Uint32 lastTick = SDL_GetTicks();
    bool quit = false;

    while (!quit) {
        Uint32 now = SDL_GetTicks();
        float deltaMs = (float)(now - lastTick);
        lastTick = now;

        // 1. イベント処理
        SDL_Event event;
        while (SDL_PollEvent(&event)) {
            if (event.type == SDL_QUIT) { quit = true; break; }
            inputManager.processEvent(event);
        }

        // 2. スクリプト実行（実行可能なら次の Op を処理）
        runner.tick();

        // 3. 状態更新（アニメーション進行）
        handler.update(deltaMs);

        // 4. 描画
        handler.render();

        // 5. フレームレート制御
        Uint32 elapsed = SDL_GetTicks() - now;
        if (elapsed < 16) SDL_Delay(16 - elapsed);
    }

    // --- 終了処理 ---
    SDL_Quit();
    return 0;
}
```

**変更のポイント:**

| Before | After |
|--------|-------|
| `while (!interpreter.isFinished()) { interpreter.step(); }` がブロック | `runner.tick()` が 1 フレーム分だけ実行して返る |
| `waitForClick()` 内に render ループ | main loop の render() が常に呼ばれる |
| `showChoices()` がブロック | 選択肢表示中も main loop が回る |
| SDL_PollEvent を直接 if 分岐 | `inputManager.processEvent(event)` |
| `SDL_Delay(16)` 固定 | 実処理時間を引いた残り時間だけ sleep |

**完了条件:**
- メインループが 1 つだけ
- waitForClick 中もパーティクル・フィルターが動く
- Ctrl+C 以外にも ESC で終了できる
